#include<stdio.h>
int main()
{
    int i,j;            // declare i as a hour an j minute
    scanf("%d%d",&i,&j);     //take input from user
    if(i==24)        //checked i=24 because it print hour 00
    {
        printf("00:00");
    }
    else if(i>12)     //because we have to convert 24 hour format into 12 hour format
    {
        i=i-12;        //to print 12 hour format
        printf("%d:%d",i,j);
    }

    else if(i==12)     //because 12 hour means 12
    {
        printf("%d:%d",i,j);
    }
    else if (i<12)
        printf("%d:%d",i,j);
}

    {
        printf("%d/%d/%d is earlier than %d/%d/%d",d2,m2,y2,d1,m1,y1);
    }
    else if(y1==y2 && m1>m2)      //if year equal then checked month
    {
        printf("%d/%d/%d is earlier than %d/%d/%d",d2,m2,y2,d1,m1,y1);
    }
    else if(y1==y2 && m1==m2 && d1>d2)      //if month year equal then checked date
    {
        printf("%d/%d/%d is earlier than %d/%d/%d",d2,m2,y2,d1,m1,y1);
    }
    else if(y2>y1)
    {

    sum=((a+c+e+g+i+k)*3+(b+d+f+h+j))-1 ;   //sum where we store 3times odd number+even numbers sum and total -1
    div=sum%10;    // where we store remainder
    sub=9-div;     //where we subtract 9 from remainder that store into  div
    printf("%d\n",sum);
    printf("%d\n",div);
    printf("%d\n",sub);
    if(sub!=l)  //checked 12 number = sub
        printf("Not valid");
    else

}
